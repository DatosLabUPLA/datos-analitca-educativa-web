IA en educación: una promesa histórica

  • A comienzos de los 80, Papert (1980, p. 189) formuló una visión temprana:

    “Proponemos enseñar IA a los niños para que ellos también puedan pensar de forma más concreta sobre los procesos mentales”.

IA en educación: una promesa histórica

  • Sin embargo, Cuban (1986) analizó las revoluciones tecnológicas en educación:
    • Radio (años 20)
    • Televisión (años 40)
    • Computadores (años 80)
  • Y detectó un ciclo recurrente:
    1. Ilusión y promesas de cambio
    2. Desilusión y resistencia
    3. Cambios lentos
    4. Nueva promesa tecnológica

IA generativa hoy: entre el optimismo y la cautela

  • Actualmente nos encontramos en la primera fase del ciclo (alta expectativa).

  • Fuerte crecimiento de investigaciones sobre IA generativa (Panqueban & Huincahue, 2024).

  • En educación secundaria se reportan mejoras en el rendimiento con IA (Xing et al., 2025).

  • Incremento de la atención visual y aceptación tecnológica (Li et al., 2025).

  • Ventajas percibidas por docentes y estudiantes (Ergene & Ergene, 2025): explicaciones detalladas, aprendizaje autónomo, resolución rápida, apoyo directo al aprendizaje

def procesar_video(video_path, output_dir):
    """
    Procesa un solo video
    Flujo:
    - Detectar cambios de pantalla por SSIM
    - Guardar pantallazos en output_dir
    - OCR directo sobre cada pantallazo
    - Clasificación de sitio (quinan, google, etc.)
    - Retorna df_final
    """

    import cv2
    import pytesseract
    import numpy as np
    import pandas as pd
    import os
    from skimage.metrics import structural_similarity as ssim
    from PIL import Image
    import re

    os.makedirs(output_dir, exist_ok=True)

    # ----------------------------
    # CONFIGURACIÓN
    # ----------------------------
    FRAME_STEP = 1
    MIN_SSIM_CHANGE = 0.90

    # ----------------------------
    # INICIALIZAR VIDEO
    # ----------------------------
    cap = cv2.VideoCapture(str(video_path))
    if not cap.isOpened():
        raise SystemExit("No se pudo abrir el video: " + str(video_path))

    fps = cap.get(cv2.CAP_PROP_FPS) or 30
    frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
    frame_interval = max(1, int(round(fps * FRAME_STEP)))
    total_duration = frame_count / fps

    # ----------------------------
    # Primer frame
    # ----------------------------
    ret, prev_frame = cap.read()
    if not ret:
        raise SystemExit("No se pudo leer el primer frame.")

    prev_gray = cv2.cvtColor(prev_frame, cv2.COLOR_BGR2GRAY)

    # guardar primera captura
    fname = f"000_t0.00.png"
    cv2.imwrite(os.path.join(output_dir, fname), prev_frame)

    records = [{
        "Segundo entrada": 0.00,
        "Segundo salida": None,
        "Tiempo en página (s)": None,
        "Captura": fname
    }]
    saved_count = 1

    # ----------------------------
    # LOOP PRINCIPAL
    # ----------------------------
    frame_idx = frame_interval

    while frame_idx < frame_count:
        cap.set(cv2.CAP_PROP_POS_FRAMES, frame_idx)
        ret, frame = cap.read()
        if not ret:
            break

        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        s = ssim(prev_gray, gray)
        prev_gray = gray

        # si hay cambio importante de pantalla
        if s < MIN_SSIM_CHANGE:
            timestamp_s = frame_idx / fps

            # cerrar registro anterior
            records[-1]["Segundo salida"] = round(timestamp_s, 2)
            records[-1]["Tiempo en página (s)"] = round(
                timestamp_s - records[-1]["Segundo entrada"], 2
            )

            # guardar nuevo pantallazo
            fname = f"{saved_count:03d}_t{timestamp_s:.2f}.png"
            cv2.imwrite(os.path.join(output_dir, fname), frame)

            records.append({
                "Segundo entrada": round(timestamp_s, 2),
                "Segundo salida": None,
                "Tiempo en página (s)": None,
                "Captura": fname
            })

            saved_count += 1

        frame_idx += frame_interval

    # cerrar último tramo
    if records:
        records[-1]["Segundo salida"] = round(total_duration, 2)
        records[-1]["Tiempo en página (s)"] = round(
            total_duration - records[-1]["Segundo entrada"], 2
        )

    cap.release()

    df = pd.DataFrame(records)

    # ----------------------------
    # OCR DIRECTO SOBRE PANTALLAZOS
    # ----------------------------
    extensiones = (".png", ".jpg", ".jpeg", ".bmp", ".tiff")
    datos = []

    for archivo in os.listdir(output_dir):
        if archivo.lower().endswith(extensiones):
            ruta_imagen = os.path.join(output_dir, archivo)
            try:
                img = Image.open(ruta_imagen)
                texto = pytesseract.image_to_string(img, lang="spa")
            except Exception as e:
                texto = f"Error al procesar: {e}"

            datos.append({
                "archivo": archivo,
                "texto_extraido": texto
            })

    df_ocr = pd.DataFrame(datos)

    # unir
    df_final = df.merge(
        df_ocr,
        left_on="Captura",
        right_on="archivo",
        how="left"
    )

    # ----------------------------
    # CLASIFICACIÓN DE SITIOS
    # ----------------------------
    categorias = {
        "quinan": ["quinan", "aulavirtual"],
        "google": ["google"],
        "chatgpt": ["chatgpt"],
        "geogebra": ["geogebra"]
    }

    def detectar_palabra(texto):
        texto = texto.lower()
        conteos = {cat: 0 for cat in categorias}

        # agrupar palabras
        for categoria, palabras in categorias.items():
            for palabra in palabras:
                conteos[categoria] += len(re.findall(rf"\b{palabra}\b", texto))

        if all(v == 0 for v in conteos.values()):
            return "otro sitio"

        categoria_max = max(conteos, key=conteos.get)
        frecuencia = conteos[categoria_max]
        return categoria_max + ("*" if frecuencia > 1 else "")

    df_final["palabra_detectada"] = df_final["texto_extraido"].apply(detectar_palabra)
    df_final["palabra_base"] = df_final["palabra_detectada"].str.replace(r"\*", "", regex=True)


    # ----------------------------
    # GUARDAR RESULTADOS CON NOMBRE DEL VIDEO
    # ----------------------------
    import os
    video_name = os.path.splitext(os.path.basename(str(video_path)))[0]
    output_csv = os.path.join(output_dir, f"resultados_{video_name}.csv")
    df_final.to_csv(output_csv, index=False)

    return df_final
